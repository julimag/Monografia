\subsection{}

\begin{frame}

   \frametitle{Design Universal, Acessível e Tecnologia Assistiva Digital}

      \begin{block}{Design Acessível}
         Produtos e serviços acessíveis e utilizáveis por pessoas com deficiência
      \end{block}

      \begin{block}{Design Universal}
         Produtos e serviços acessíveis e utilizáveis por todos os indivíduos, independente de possuírem ou não deficiência. \\ 
         Beneficiar todas as idades e capacidades, sem discriminações.
      \end{block}
      
      \begin{block}{Tecnologia Assistiva Digital}
         Refere-se aos sistemas que oferecem soluções para tentar suprir as limitações de uma parcela da sociedade.
      \end{block}

\end{frame}

\subsection{}

\begin{frame}

   \frametitle{Acessibilidade e Usabilidade Digitais}

      \begin{block}{Usabilidade}
	Usabilidade é sinônimo de facilidade de uso! É algo relativo!
	\begin{itemize}
	\item Manuseio fácil, rapidamente aprendido, dificilmente esquecido, não provoca erros operacionais
	\item Alto grau de satisfação e resolve eficientemente as tarefas para os quais foi projetado
	\end{itemize}
      \end{block}

      \begin{block}{Acessibilidade}
	Termo para indicar a possibilidade de qualquer pessoa usufruir de todos os benefícios disponíveis, independente de deficiência.
      \end{block}
      
      \begin{block}{Acessibilidade X Usabilidade}
	\begin{itemize}
	 \item Software orientado à usabilidade não é necessariamente orientado à usabilidade!
	 \item O software pode ser fácil para usuário comuns mas inacessível para aqueles com necessidades especiais!
	\end{itemize}
      \end{block}

\end{frame}

	\item Validação: Deve-se considerar as habilidades específicas; a interação harmoniosa do usuário também é um requisito!

	\begin{itemize}
	 \item W3C: Criou o WAI (Web Acessibility Initiative)
	  \begin{itemize}
	  \item Lançou o WCAG 1.0 e o 2.0 (Diretrizes para a Acessibilidade do Conteúdo da Web)
	  \end{itemize}
	  \item Brasil
	    \begin{itemize}
	    \item Lei 10.098 (2000) - normas e critérios para acessibilidade
	    \item Decreto 5.296 (2004) - acessibilidade na internet
	    \item Comitê Brasileiro (CB) da ABNT - normatização no campo da acessibilidade
	    \item eMAG - Modelo de Acessibilidade do Governo Eletrônico
	    \item Selo de Acessibilidade Digital - aferição da política de acessibilidade dos sites
	    \end{itemize}
	\end{itemize}
      \end{block}

\end{frame}

\subsection{Acessibilidade na Web}

\begin{frame}

   \frametitle{Acessibilidade na Web}

      \begin{block}{Características e Importância}	
	\begin{itemize}
	 \item Acesso às informações (sem prejuízos de conteúdo) independentemente de:
	    \begin{itemize}
	    \item Características técnicas dos equipamentos (ex: infra-estrutura precária, ambiente desfavorável) 
	    \item Características corporais (ex: deficiências sensoriais, motora)
	    \end{itemize}
	 \item Importância para os deficientes
	    \begin{itemize}
	    \item Novas formas de relacionamento
	    \item Alternativas de diversão
	    \item Qualificação (oportunidades de trabalho)
	    \item Acesso a informações antes obtidas através de terceiros
	    \end{itemize}	 
	\end{itemize}
      \end{block}

\end{frame}

%===============================================================================
% Proposta
%-------------------------------------------------------------------------------
% Acessibilidade na Web
%===============================================================================

\begin{frame}

   \frametitle{Acessibilidade na Web}

      \begin{block}{Dificuldade}
	\begin{itemize}
	 \item Como satisfazer as necessidades de acesso dos usuários? Usuário web pode ser qualquer um!
	\end{itemize}
      \end{block}
      
      \begin{block}{Solução}
	\begin{itemize}
	 \item Delimitar um grupo-alvo para foco do desenvolvimento (maior público possível)
	 \item Apresentação da informação de formas múltiplas (redundância)
	 \item Verificação e validação devem considerar os mínimos quesitos de acessibilidade
	\end{itemize}
      \end{block}
      
      Site totalmente acessível é utopia! Mas o desenvolvimento deve ser voltado, pelo menos, para atender padrões mínimos de acessibilidade.
      
\end{frame}

\subsection{Metodologia Ágil: Simplicidade e Participação do Cliente}

\begin{frame}

   \frametitle{Metodologia Ágil: Simplicidade e Participação do Cliente}

      \begin{block}{Simplicidade}
         \begin{itemize}
          \item Um dos princípios do manifesto ágil
          \item Evitar complicações, artifícios, extravagâncias e excessos
          \item Mais favorável a mudanças (mais fácil incluir/remover/alterar algo)
          \item Não é apenas estética, nem design simplório, nem redução de funcionalidades
          \item É foco na criação de real valor para o cliente, é facilidade de uso
         \end{itemize}         
      \end{block}

      \begin{block}{Participação do Cliente}
         \begin{itemize}
          \item Maior comunicação, feedback mais rápido
          \item Melhor compreensão
	    \begin{itemize}
	      \item dos requisitos pelos desenvolvedores
	      \item do processo de desenvolvimento pelo cliente
	    \end{itemize}
          \item Produtos mais próximos da verdadeira realidade do usuário
          \item Detecção precoce de falhas e correção prematura
         \end{itemize}
      \end{block}
      
\end{frame}

\subsection{Metodologia Ágil e a Participação do Usuário Deficiente}

\begin{frame}

   \frametitle{Metodologia Ágil e a Participação do Usuário Deficiente}

      \begin{block}{}
         \begin{itemize}
          \item Software acessível não é atender apenas a lista de recomendações de acessibilidade (instruções/guia de orientação)
          \item Conhecer as necessidades especiais e as preferências do usuário
          \item O deficiente é o usuário! O deficiente conhece suas necessidades! Ele detecta as falhas!
          \item Desenvolvimento ágil prega participação do cliente, neste caso, do deficiente
         \end{itemize}
      \end{block}

      \begin{block}{Valores e Princípios do Manifesto Ágil}
         \begin{itemize}
          \item Colaboração do cliente mais que negociação de contratos
          \item Pessoas relacionadas à negócios e desenvolvedores devem trabalhar em conjunto e diariamente, durante todo o curso do projeto
          \item O método mais eficiente e eficaz de transmitir informações para, e por dentro de um time de desenvolvimento, é através de uma conversa cara a cara
         \end{itemize}
      \end{block}
      
\end{frame}

\begin{frame}

   \frametitle{Potenciais Obstáculos entre o Software e o Deficiente}

      Na maioria das vezes, no desenvolvimento do software, não há a preocupação com acessibilidade.
      
      \begin{block}{Algumas dificuldades p/ deficiente}
         \begin{itemize}
          \item Visual: Imagens sem descrição textual (p/ leitores de tela), vídeos sem descrição sonora
          \item Auditiva: Ausência de legendas para áudio
          \item Motora: Páginas de difícil navegabilidade
          \item Mental: Falta de clareza e organização
         \end{itemize}
         
      \end{block}

\end{frame}

%===============================================================================
% Análise
%-------------------------------------------------------------------------------
% Potenciais Obstáculos entre Acessibilidade e os Métodos Ágeis
%===============================================================================

\begin{frame}

   \frametitle{Potenciais Obstáculos entre Acessibilidade e os Métodos Ágeis}

      \begin{block}{Dificuldades}
         \begin{itemize}
          \item Foco e motivação da equipe: Foco apenas em medir o progresso do projeto e Motivação concentrada apenas na qualidade de código
          \item Perfil do cliente: pode ser álguem que não representa as necessidades comuns de todos os usuários
          \item Levantamento de requisitos: estórias (XP) e lista priorizada (Scrum) podem não contemplar os requisitos de acessibilidade
          \item Construção das interfaces: mudanças podem impactar em interfaces de ciclos anteriores e tornar-se um fator complicador
          \item Testes: inexistência de práticas específicas para avaliar acessibilidade durante o desenvolvimento ágil
         \end{itemize}

      \end{block}

\end{frame}

%===============================================================================
% Análise
%-------------------------------------------------------------------------------
% Análise do Ágil com Acessibilidade
%===============================================================================

\begin{frame}

   \frametitle{Análise do Ágil com Acessibilidade}

      \begin{block}{Porque prover Acessibilidade?}
	 Produto com sucesso requer qualidade!
         \begin{itemize}
          \item Qualidade de uso
          \begin{itemize}
           \item Está relacionado com capacidade e facilidade dos usuários atingirem metas com eficiência e satisfação.
           \item Aumento ou queda de produtividade dos usuários
           \item Aumento ou redução de custos com manutenção
          \end{itemize}
          \item Rentabilidade: ampliação da clientela (novo público - satisfação e fidelidade)
         \end{itemize}
      \end{block}

      \begin{block}{Porque Métodos Ágeis?}
         \begin{itemize}
          \item O Ágil prega participação/colaboração, simplicidade, motivação
         \end{itemize}
      \end{block}

\end{frame}

\section{Conclusão}

\begin{frame}

   \frametitle{Desenvolvimento ágil orientado à acessibilidade}

      \begin{block}{O que falta p/ fomentar?}
         \begin{itemize}
          \item Maior comprometimento com a acessibilidade pelos envolvidos
          \item Percepção da rentabilidade de investimentos em acessibilidade
         \end{itemize}
      \end{block}

      \begin{block}{O que ajuda?}
         \begin{itemize}
          \item Capacitação dos envolvidos
          \item Estratégia de integração entre metodologias ágeis e requisitos de acessibilidade
          \item Conscientização da sociedade e do poder público em relação à acessibilidade
         \end{itemize}
      \end{block}
      
\end{frame}

% Conclusão

\begin{frame}

   \frametitle{Ágil com Acessibilidade}

      \begin{block}{Porque?}
         \begin{itemize}
          \item Ágil prega simplicidade, participação do usuário, qualidade
          \item Simplicidade, participação do usuário (deficiente), qualidade favorecem a acessibilidade
         \end{itemize}
         A escolha de medologia ágil para o desenvolvimento de software com design acessível (ou até universal), ao invés do tradicional,
         é conveniente e adequada!
      \end{block}

\end{frame}
